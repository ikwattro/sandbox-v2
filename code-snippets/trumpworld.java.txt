import org.neo4j.driver.v1.*;
import static org.neo4j.driver.v1.Values.parameters;

import java.util.List;
import static java.util.Arrays.asList;
import static java.util.Collections.singletonMap;

public class TrumpworldSandbox {

    public static void main(String...args) {
        Config noSSL = Config.build().withEncryptionLevel(Config.EncryptionLevel.NONE).toConfig();
        Driver driver = GraphDatabase.driver("bolt://<HOST>:<BOLTPORT>",AuthTokens.basic("<USERNAME>","<PASSWORD>"),noSSL); // <password>
        try (Session session = driver.session()) {
            // what are the organizations in Trumpworld?
            String cypherQuery =
                "MATCH (o:Organization) " + 
                "RETURN o.name AS name " +
                "LIMIT $limit";
            StatementResult result = session.run(cypherQuery, parameters("limit", 10));
            while (result.hasNext()) {
                System.out.println(result.next().get("name"));
            }
        }
    }
}
